int bfnn_point(CloudPtr cloud, const Vec3f& point) {
    return std::min_element(cloud->points.begin(), cloud->points.end(),
        [&point](const PointType& pt1, const PointType& pt2) -> bool {
        return (pt1.getVector3fMap() - point).squaredNorm() <

              }) -cloud->points.begin();
}
