template <typename C, typename D, typename Getter>
void ComputeMeanAndCovDiag(const C& data, D& mean, D& cov_diag, Getter&& getter) {
    size_t len = data.size();
    assert(len > 1);
    // clang-format off
    //é€šè¿‡getterè·å–æ¯ä¸ªç´¢å¼•å¯¹åº”çš„åæ ‡ï¼Œå¹¶è®¡ç®—å‡å€¼ä¸åæ–¹å·®
    mean = std::accumulate(data.begin(), data.end(), D::Zero().eval(),
                           [&getter](const D& sum, const auto& data) -> D { return sum + getter(data); }) / len;
    cov_diag = std::accumulate(data.begin(), data.end(), D::Zero().eval(),
                               [&mean, &getter](const D& sum, const auto& data) -> D {
                                   return sum + (getter(data) - mean).cwiseAbs2().eval();
                               }) / (len - 1);
    // clang-format on
}

    const auto create_if_not_empty = [&node, this](KdTreeNode *&new_node, const IndexVec &index) {

    bool FindSplitAxisAndThresh(const IndexVec& point_idx, int& axis, float& th, IndexVec& left, IndexVec& right){

    Vec3f mean;
    math::ComputeMeanAndCovDiag(point_idx, mean, var, [this](int idx) { return cloud_[idx]; });//é€šè¿‡lambdaè¡¨è¾¾å¼ï¼Œå°†ç´¢å¼•å¯¹åº”çš„ç‚¹ä¼ ç»™å‡½æ•°

    var.maxCoeff(&max_i, &max_j);
    axis = max_i;//æœ€å¤§åæ–¹å·®è½´
    th = mean[axis]; //é˜ˆå€¼

    for (const auto &idx : point_idx) {
        if (cloud_[idx][axis] < th) {
            // ä¸­ä½æ•°å¯èƒ½å‘å·¦å–æ•´
            left.emplace_back(idx);
        } else {
            right.emplace_back(idx);
        }
    }

    // è¾¹ç•Œæƒ…å†µæ£€æŸ¥ï¼šè¾“å…¥çš„pointsç­‰äºåŒä¸€ä¸ªå€¼ï¼Œä¸Šé¢çš„åˆ¤å®šæ˜¯>=å·ï¼Œæ‰€ä»¥éƒ½è¿›äº†å³ä¾§
    // è¿™ç§æƒ…å†µä¸éœ€è¦ç»§ç»­å±•å¼€ï¼Œç›´æ¥å°†å½“å‰èŠ‚ç‚¹è®¾ä¸ºå¶å­å°±è¡Œ
    if (point_idx.size() > 1 && (left.empty() || right.empty())) {
        return false;
    }

    return true;
    }

    void Reset(){
        tree_node_id_ = 0;
        root_.reset(new KdTreeNode());
        root_->id_ = tree_node_id_++;
        size_ = 0;
    }

    /// ä¸¤ä¸ªç‚¹çš„å¹³æ–¹è·ç¦»
    static inline float Dis2(const Vec3f& p1, const Vec3f& p2) { return (p1 - p2).squaredNorm(); }

           if (node->IsLeaf()) {
        // å¦‚æœæ˜¯å¶å­ï¼Œæ£€æŸ¥å¶å­æ˜¯å¦èƒ½æ’å…¥

    // çœ‹ptè½åœ¨å·¦è¿˜æ˜¯å³ï¼Œä¼˜å…ˆæœç´¢ptæ‰€åœ¨çš„å­æ ‘
    // ç„¶åå†çœ‹å¦ä¸€ä¾§å­æ ‘æ˜¯å¦éœ€è¦æœç´¢

        that_side = node->right_;

        this_side = node->right_;
        that_side = node->left_;

    //åˆ¤æ–­æ˜¯å¦éœ€è¦å±•å¼€
    if (NeedExpand(pt, node, knn_result)) {  // æ³¨æ„è¿™é‡Œæ˜¯è·Ÿè‡ªå·±æ¯”
        Knn(pt, that_side, knn_result);

    }

    /**
     * å¯¹å¶å­èŠ‚ç‚¹ï¼Œè®¡ç®—å®ƒå’ŒæŸ¥è¯¢ç‚¹çš„è·ç¦»ï¼Œå°è¯•æ”¾å…¥ç»“æœä¸­

    void ComputeDisForLeaf(const Vec3f& pt, KdTreeNode* node, std::priority_queue<NodeAndDistance>& result) const{

    }

     * @param node Kdtree èŠ‚ç‚¹

    bool NeedExpand(const Vec3f& pt, KdTreeNode* node, std::priority_queue<NodeAndDistance>& knn_result) const{
            if (knn_result.size() < k_) {
        return true;
    }
    //å‰é¢ä¹Ÿè¯´è¿‡,åˆ¤æ–­å¦ä¸€ä¾§çš„æ˜¯å¦å±•å¼€çš„æ¡ä»¶æ˜¯åˆ¤æ–­å½“å‰æ‰¾åˆ°æœ€å°çš„è·ç¦»æ˜¯å¦å°äºåˆ°åˆ†ç•Œçº¿çš„è·ç¦»ï¼Œ
    //å¦‚æœæ˜¯ï¼Œåˆ™è¯´æ˜å¦ä¸€ä¾§çš„ç‚¹è‚¯å®šæ˜¯æ›´è¿œçš„ï¼Œåˆ™ä¸ç”¨å±•å¼€
    if (approximate_) {
        float d = pt[node->axis_index_] - node->split_thresh_;
        if ((d * d) < knn_result.top().distance2_ * alpha_) {
            return true;
        } else {
            return false;
        }
    } else {
        // æ£€æµ‹åˆ‡é¢è·ç¦»ï¼Œçœ‹æ˜¯å¦æœ‰æ¯”ç°åœ¨æ›´å°çš„
        float d = pt[node->axis_index_] - node->split_thresh_;
        if ((d * d) < knn_result.top().distance2_) {
            return true;
        } else {
            return false;
        }
    }
    }

    int k_ = 5;                                   // knnæœ€è¿‘é‚»æ•°é‡
    std::shared_ptr<KdTreeNode> root_ = nullptr;  // æ ¹èŠ‚ç‚¹
    std::vector<Vec3f> cloud_;                    // è¾“å…¥ç‚¹äº‘
    std::unordered_map<int, KdTreeNode*> nodes_;  // for bookkeeping

    size_t size_ = 0;       // å¶å­èŠ‚ç‚¹æ•°é‡
    int tree_node_id_ = 0;  // ä¸ºkdtree node åˆ†é…id

    bool approximate_ = true;
